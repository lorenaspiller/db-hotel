GROUP BY
1) Conta gli ospiti raggruppandoli per anno di nascita
	SELECT COUNT(`id`) AS 'ospiti_per_anno', YEAR(`date_of_birth`)
	FROM `ospiti`
	GROUP BY YEAR(`date_of_birth`);

2) Somma i prezzi dei pagamenti raggruppandoli per status
	SELECT SUM(`price`) AS 'totale_prezzi', `status`
	FROM `pagamenti`
	GROUP BY `status`;
	
3) Conta quante volte è stata prenotata ogni stanza
	SELECT COUNT(`id`) AS 'numero-prenotazioni', `stanza_id` AS 'stanza'
	FROM `prenotazioni`
	GROUP BY `stanza`;
	
4) Fai una analisi per vedere se ci sono ore in cui le prenotazioni sono più frequenti
	SELECT COUNT(`id`) AS 'numero_prenotazioni', HOUR(`created_at`) AS 'orario_medio'
	FROM `prenotazioni`
	GROUP BY HOUR(`created_at`)
	ORDER BY COUNT(`id`) DESC;

5) Quante prenotazioni ha fatto l’ospite che ha fatto più prenotazioni?
	SELECT COUNT(`prenotazione_id`) AS 'numero_prenotazioni', `ospite_id` AS 'ospite'
	FROM `prenotazioni_has_ospiti`
	GROUP BY `ospite`
	ORDER BY `numero_prenotazioni`  DESC
	LIMIT 1;
