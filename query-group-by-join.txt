GROUP BY
1) Conta gli ospiti raggruppandoli per anno di nascita
	SELECT COUNT(`id`) AS 'ospiti_per_anno', YEAR(`date_of_birth`)
	FROM `ospiti`
	GROUP BY YEAR(`date_of_birth`);

2) Somma i prezzi dei pagamenti raggruppandoli per status
	SELECT SUM(`price`) AS 'totale_prezzi', `status`
	FROM `pagamenti`
	GROUP BY `status`;
	
3) Conta quante volte è stata prenotata ogni stanza
	SELECT COUNT(`id`) AS 'numero-prenotazioni', `stanza_id` AS 'stanza'
	FROM `prenotazioni`
	GROUP BY `stanza`;
	
4) Fai una analisi per vedere se ci sono ore in cui le prenotazioni sono più frequenti
	SELECT COUNT(`id`) AS 'numero_prenotazioni', HOUR(`created_at`) AS 'orario_medio'
	FROM `prenotazioni`
	GROUP BY HOUR(`created_at`)
	ORDER BY COUNT(`id`) DESC;

5) Quante prenotazioni ha fatto l’ospite che ha fatto più prenotazioni?
	SELECT COUNT(`prenotazione_id`) AS 'numero_prenotazioni', `ospite_id` AS 'ospite'
	FROM `prenotazioni_has_ospiti`
	GROUP BY `ospite`
	ORDER BY `numero_prenotazioni`  DESC
	LIMIT 1;

1) Come si chiamano gli ospiti che hanno fatto più di due prenotazioni?
	SELECT `name`, `lastname`, `ospite_id`, COUNT(`prenotazione_id`) AS 'numero_prenotazioni'
	FROM `ospiti` 
	INNER JOIN `prenotazioni_has_ospiti`
	ON `prenotazioni_has_ospiti`.`ospite_id` = `ospiti`.`id`
	GROUP BY `ospite_id`
	HAVING `numero_prenotazioni` > 2;

3) Stampare Nome, Cognome, Prezzo e Pagante per tutte le prenotazioni fatte a Maggio 2018
	SELECT `ospiti`.`name` AS 'NOME_OSPITE', `ospiti`.`lastname` AS 'COGNOME_OSPITE', `pagamenti`.`price` AS 'PREZZO', `paganti`.`name` AS 'NOME_PAGANTE',  `paganti`.`lastname` AS 'COGNOME_PAGANTE', `prenotazioni`.`created_at` AS 'PRENOTAZIONI_FATTE'
	FROM `ospiti`
	LEFT JOIN `paganti`
	ON `paganti`.`ospite_id` = `ospiti`.`id`
	LEFT JOIN `pagamenti`
	ON `pagamenti`.`pagante_id` = `paganti`.`id`
	LEFT JOIN `prenotazioni`
	ON `pagamenti`.`prenotazione_id` = `prenotazioni`.`id`  
	WHERE Month(`prenotazioni`.`created_at`)='05' && YEAR(`prenotazioni`.`created_at`)='2018';
	
	oppure 
	WHERE EXTRACT( YEAR_MONTH FROM `prenotazioni`.`created_at` ) = '201805'

4) Fai la somma di tutti i prezzi delle prenotazioni per le stanze del primo piano
	SELECT SUM(`price`) AS 'TOTALE_PRENOTAZIONI_PRIMO_PIANO'
	FROM `pagamenti`
	INNER JOIN `prenotazioni`
	ON `pagamenti`.`prenotazione_id` = `prenotazioni`.`id`
	INNER JOIN `stanze`
	ON `prenotazioni`.`stanza_id` = `stanze`.`id`
	WHERE `stanze`.`floor` = 1;

5) Prendi i dati di fatturazione per la prenotazione con id=7
	SELECT `prenotazioni`.`id`, `name`, `lastname`, `address` 
	FROM `paganti`
	INNER JOIN `pagamenti`
	ON `pagamenti`.`pagante_id` = `paganti`.`id`
	INNER JOIN `prenotazioni`
	ON `prenotazioni`.`id` = `pagamenti`.`prenotazione_id`
	WHERE `prenotazioni`.`id` = 7;

6) Le stanze sono state tutte prenotate almeno una volta? (Visualizzare le stanze non ancora prenotate) 
	SELECT `room_number`, `prenotazioni`.`stanza_id`
	FROM `stanze`
	LEFT JOIN `prenotazioni`
	ON `prenotazioni`.`stanza_id` = `stanze`.`id`
	WHERE `prenotazioni`.`stanza_id` IS NULL;
